package chapter02;

public class Solution01 {
    //直接遍历整个数组，找出target
    public boolean searchTarget(int[][] array,int target){
        for(int[] row:array){
            for(int element:row ){
                if (element==target){
                return true;
                }
            }
        }
        return false;
    }
}

package chapter02;

public class Solution02 {

public boolean searchTargetOne(int[][] array,int target){
    for(int[] row:array){
        int index = search(row,target);
        if(index>0){
            return true;
        }
    }
    return false;
}
    public int search(int[] nums,int target){
        int left=0,right=nums.length-1;
        while (left<=right){
            int middle=(left+right)/2;
            if(target == nums[middle]){
                return nums[middle];
            } else if (target < nums[middle]) {
                right=middle-1;
            } else if (target > nums[middle]) {
                left=middle+1;
            }
        }
        return -1;
    }


    public boolean searchTargetTwo(int[][] array, int target) {
        if (array == null || array.length == 0) {
            return false;
        }
        int left = 0;
        int right = array.length * array[0].length - 1;
        int col = array[0].length;
        while (left <= right) {
            int mid = (left + right) / 2;
            int value = array[mid / col][mid % col];
            if (value == target) { return true;
            }
            else if (value < target) {
                left = mid + 1;
            }
            else { right = mid - 1;
            }
        }
        return false;
    }
}
